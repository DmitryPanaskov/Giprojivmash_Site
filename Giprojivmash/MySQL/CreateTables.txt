DROP TABLE IF EXISTS page;

DROP TABLE IF EXISTS contactdata;
DROP TABLE IF EXISTS contact;

DROP TABLE IF EXISTS historyphoto;
DROP TABLE IF EXISTS history;

DROP TABLE IF EXISTS portfoliophoto;
DROP TABLE IF EXISTS portfolio;

DROP TABLE IF EXISTS servicethirdlayer;
DROP TABLE IF EXISTS servicesecondlayer;
DROP TABLE IF EXISTS servicefirstlayer;

DROP TABLE IF EXISTS vacancy;

CREATE TABLE page(
    Id int NOT NULL AUTO_INCREMENT,
    PageName varchar(50),
    Title varchar(500),
    Keyword varchar(500),
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE contact(
    Id int NOT NULL AUTO_INCREMENT,
    Photo varchar(50),
    Description varchar(500),
    FirstName varchar(500),
    LastName varchar(500),
    Patronymic varchar(500),
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE contactdata(
    Id int NOT NULL AUTO_INCREMENT,
    ContactId int NOT NULL,
    Data varchar(50),
    SubData varchar(50),
    Type int,
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE history(
    Id int NOT NULL AUTO_INCREMENT,
    Description varchar(5000),
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE historyphoto(
    Id int NOT NULL AUTO_INCREMENT,
    HistoryId int NOT NULL,
    PhotoName varchar(50),
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE portfolio(
    Id int NOT NULL AUTO_INCREMENT,
    Description varchar(5000),
    PRIMARY KEY (Id));

CREATE TABLE portfoliophoto(
    Id int NOT NULL AUTO_INCREMENT,
    PortfolioId int NOT NULL,
    PhotoName varchar(50),
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE servicefirstlayer(
    Id int NOT NULL AUTO_INCREMENT,
    Description varchar(1000),
    Title varchar(500),
    PhotoTitle varchar(50),
    DescriptionShort varchar(500),
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
    
CREATE TABLE servicesecondlayer(
    Id int NOT NULL AUTO_INCREMENT,
    ServiceFirstLayerId int NOT NULL,
    Description varchar(1000),
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
    
CREATE TABLE servicethirdlayer(
    Id int NOT NULL AUTO_INCREMENT,
    ServiceSecondLayerId int NOT NULL,
    Description varchar(1000),
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE vacancy(
    Id int NOT NULL AUTO_INCREMENT,
    Position varchar(500),
    Description varchar(1000),
    Contact varchar(500),
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
