DROP TABLE IF EXISTS contactdata;
DROP TABLE IF EXISTS contact;

DROP TABLE IF EXISTS historyphoto;
DROP TABLE IF EXISTS history;

DROP TABLE IF EXISTS portfoliophoto;
DROP TABLE IF EXISTS portfolio;

DROP TABLE IF EXISTS vacancy;

/*Identity*/
DROP TABLE IF EXISTS AspNetRoleClaims;
DROP TABLE IF EXISTS AspNetRoles;
DROP TABLE IF EXISTS AspNetUserClaims;
DROP TABLE IF EXISTS AspNetUserLogins;
DROP TABLE IF EXISTS AspNetUserRoles;
DROP TABLE IF EXISTS AspNetUsers;
DROP TABLE IF EXISTS AspNetUserTokens;

CREATE TABLE contact(
    Id int NOT NULL AUTO_INCREMENT,
    Position varchar(50),
    Photo varchar(50),
    Description varchar(450),
    FirstName varchar(256),
    LastName varchar(256),
    Patronymic varchar(450),
    PositionType int,
    IndexNumber int,
    Gender int,
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE contactdata(
    Id int NOT NULL AUTO_INCREMENT,
    ContactId int NOT NULL,
    Data varchar(50),
    SubData varchar(50),
    ContactDataType int,
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE history(
    Id int NOT NULL AUTO_INCREMENT,
    Description varchar(450),
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE historyphoto(
    Id int NOT NULL AUTO_INCREMENT,
    HistoryId int NOT NULL,
    PhotoName varchar(50),
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE portfolio(
    Id int NOT NULL AUTO_INCREMENT,
    Description varchar(450),
    PRIMARY KEY (Id));

CREATE TABLE portfoliophoto(
    Id int NOT NULL AUTO_INCREMENT,
    PortfolioId int NOT NULL,
    PhotoName varchar(50),
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE vacancy(
    Id int NOT NULL AUTO_INCREMENT,
    Position varchar(450),
    Description varchar(1000),
    PhoneNubmer varchar(50),
    SubPhoneNubmer varchar(50),
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

     /*Identity*/
CREATE TABLE AspNetRoleClaims(
    Id int NOT NULL AUTO_INCREMENT,
    ClaimType varchar(450),
    ClaimValue varchar(450),
    RoleId varchar(450) NOT NULL,
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE AspNetRoles(
    Id varchar(256) NOT NULL,
    ConcurrencyStamp text,
    Name varchar(256),
    NormalizedName varchar(256),
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE AspNetUserClaims(
    Id int NOT NULL AUTO_INCREMENT,
    ClaimType text,
    ClaimValue text,
    UserId varchar(450) NOT NULL,
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE AspNetUserLogins(
    LoginProvider varchar(450) NOT NULL,
    ProviderKey varchar(450) NOT NULL,
    ProviderDisplayName text,
    UserId varchar(450) NOT NULL)
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE AspNetUserRoles(
    UserId varchar(256) NOT NULL,
    RoleId varchar(256) NOT NULL)
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE AspNetUsers(
	Id varchar(256) NOT NULL,
	AccessFailedCount int NOT NULL,
    Discriminator text,
	ConcurrencyStamp text,
	Email varchar(256),
	EmailConfirmed bit NOT NULL,
	LockoutEnabled bit NOT NULL,
	LockoutEnd timestamp,
	NormalizedEmail varchar(256),
	NormalizedUserName varchar(256),
	PasswordHash text,
	PhoneNumber text,
	PhoneNumberConfirmed bit NOT NULL,
	SecurityStamp text,
	TwoFactorEnabled bit NOT NULL,
	UserName varchar(256),
    PRIMARY KEY (Id))
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE AspNetUserTokens(
	UserId varchar(256) NOT NULL,
	LoginProvider varchar(450) NOT NULL,
	Name varchar(450) NOT NULL,
	Value text)
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
